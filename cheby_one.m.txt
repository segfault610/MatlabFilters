Fs = 1000;
t = 0:1/Fs:2;
% Same signal
x = sin(2*pi*5*t) + 0.5*sin(2*pi*100*t);
% Chebyshev Type I
n = 4;
Rp = 1;         % Passband ripple
Wn = 20/(Fs/2); % Cutoff frequency
[b, a] = cheby1(n, Rp, Wn);
y = filter(b, a, x);
figure;
subplot(2,1,1); plot(t,x); title('Original Signal');
subplot(2,1,2); plot(t,y); title('Chebyshev Filtered Signal');
